    <div class="tableCell">

    </div>

.table {
    height: 200px;/*高度值不能少*/
    width: 200px;/*宽度值不能少*/

    position: relative;
    float:left;
    background: yellow;
}      
 
.tableCell {
    display: table-cell;
    vertical-align: middle;
    text-align: center;        
    *position: absolute;
    padding: 10px;
    *top: 50%;
    *left: 50%;
}

    *position:relative;
    *top: -50%;
    *left: -50%;
     background: green;
